Slutsky's theorem:\\
    If 
\begin{equation}
    \label{asmp1}
    X_n \convg{D} X
\end{equation}
    and
\begin{equation}
    \label{asmp2}
    Z_n \convg{P} z_0
\end{equation}
for some constant $z_0\in \mathbb{R}$, then:
\begin{enumerate}[a)]
    \item Prove:  
        \label{Sl_add}
        $
        X_n + Z_n \overset{D}{\rightarrow} X + z_0
        $
        \\
        \input{contents/q1_a.tex}
    \item Prove:
        \label{Sl_mult}
        $
        X_n \cdot Z_n \overset{D}{\rightarrow} X \cdot z_0
        $
        \\
        \input{contents/q1_b.tex}
    \item Prove:
        \label{Sl_div}
        $
        X_n / Z_n \overset{D}{\rightarrow} X /  z_0, \quad z_0 \neq 0
        $
        \\
        \input{contents/q1_c.tex}
\end{enumerate}
